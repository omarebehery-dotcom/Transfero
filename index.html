import 'dart:io'; import 'package:flutter/material.dart'; import 'package:nearby_connections/nearby_connections.dart'; import 'package:qr_flutter/qr_flutter.dart'; import 'package:mobile_scanner/mobile_scanner.dart'; // --- الصفحة الرئيسية --- class MainMenu extends StatelessWidget { @override Widget build(BuildContext context) { return Scaffold( appBar: AppBar(title: Text("ShareIt Clone"), centerTitle: true), body: Center( child: Column( mainAxisAlignment: MainAxisAlignment.center, children: [ Icon(Icons.share_arrival_time, size: 80, color: Colors.blue), SizedBox(height: 50), ElevatedButton.icon( icon: Icon(Icons.send), label: Text("إرسال ملف"), style: ElevatedButton.styleFrom(minimumSize: Size(200, 60)), onPressed: () => Navigator.push(context, MaterialPageRoute(builder: (c) => SendScreen())), ), SizedBox(height: 20), ElevatedButton.icon( icon: Icon(Icons.call_received), label: Text("استلام ملف"), style: ElevatedButton.styleFrom(minimumSize: Size(200, 60), backgroundColor: Colors.green.shade100), onPressed: () => Navigator.push(context, MaterialPageRoute(builder: (c) => ReceiveScreen())), ), ], ), ), ); } } // --- صفحة الإرسال --- class SendScreen extends StatefulWidget { @override _SendScreenState createState() => _SendScreenState(); } class _SendScreenState extends State { String? selectedFilePath; double progress = 0.0; bool connecting = false; Future selectFile() async { // طلب صلاحيات الكاميرا والموقع والملفات await [Permission.camera, Permission.location, Permission.storage, Permission.nearbyWifiDevices].request(); FilePickerResult? result = await FilePicker.platform.pickFiles(); if (result != null) { setState(() => selectedFilePath = result.files.single.path); } } @override Widget build(BuildContext context) { return Scaffold( appBar: AppBar(title: Text("إرسال")), body: Column( children: [ if (selectedFilePath == null) Expanded(child: Center(child: ElevatedButton(onPressed: selectFile, child: Text("اختر الملف أولاً")))) else if (!connecting) Expanded( child: Stack( children: [ MobileScanner( onDetect: (capture) { final List barcodes = capture.barcodes; if (barcodes.isNotEmpty) { startConnection(barcodes.first.rawValue!); } }, ), Center(child: Container(width: 250, height: 250, decoration: BoxDecoration(border: Border.all(color: Colors.white)))), Positioned(top: 20, left: 0, right: 0, child: Center(child: Text("امسح كود المستلم", style: TextStyle(color: Colors.white, fontSize: 18, backgroundColor: Colors.black45)))) ], ), ) else Expanded( child: Center( child: Padding( padding: const EdgeInsets.all(20.0), child: Column( mainAxisAlignment: MainAxisAlignment.center, children: [ Text("جاري نقل الملف: ${selectedFilePath!.split('/').last}"), SizedBox(height: 20), LinearProgressIndicator(value: progress, minHeight: 10), SizedBox(height: 10), Text("${(progress * 100).toStringAsFixed(1)} %"), ], ), ), ), ), ], ), ); } void startConnection(String endpointId) async { setState(() => connecting = true); Nearby().requestConnection("SenderDevice", endpointId, onConnectionInitiated: (id, info) => Nearby().acceptConnection(id, onPayLoadRecieved: (id, payload) {}), onConnectionResult: (id, status) { if (status == Status.CONNECTED) { Nearby().sendPayload(Payload.forFile(File(selectedFilePath!)), id); } }, onDisconnected: (id) => setState(() => connecting = false), ); Nearby().onPayloadTransferUpdate = (id, update) { if (update.status == PayloadStatus.IN_PROGRESS) { setState(() => progress = (update.bytesTransferred / update.totalBytes)); } else if (update.status == PayloadStatus.SUCCESS) { ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text("تم الإرسال!"))); Navigator.pop(context); } }; } } // --- صفحة الاستقبال --- class ReceiveScreen extends StatefulWidget { @override _ReceiveScreenState createState() => _ReceiveScreenState(); } class _ReceiveScreenState extends State { String myId = "User_${DateTime.now().millisecond}"; double progress = 0.0; bool isReceiving = false; @override void initState() { super.initState(); startDiscovery(); } void startDiscovery() async { await [Permission.location, Permission.nearbyWifiDevices].request(); Nearby().startAdvertising(myId, Strategy.P2P_STAR, onConnectionInitiated: (id, info) => Nearby().acceptConnection(id, onPayLoadRecieved: (id, payload) { if (payload.type == PayloadType.FILE) setState(() => isReceiving = true); }), onConnectionResult: (id, status) => print(status), onDisconnected: (id) => print("Disconnected"), ); Nearby().onPayloadTransferUpdate = (id, update) { if (update.status == PayloadStatus.IN_PROGRESS) { setState(() => progress = (update.bytesTransferred / update.totalBytes)); } else if (update.status == PayloadStatus.SUCCESS) { ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text("استلمت الملف!"))); setState(() => isReceiving = false); } }; } @override void dispose() { Nearby().stopAdvertising(); super.dispose(); } @override Widget build(BuildContext context) { return Scaffold( appBar: AppBar(title: Text("استقبال")), body: Center( child: isReceiving ? Column( mainAxisAlignment: MainAxisAlignment.center, children: [ Text("جاري الاستلام..."), Padding( padding: const EdgeInsets.all(20.0), child: LinearProgressIndicator(value: progress), ), Text("${(progress * 100).toStringAsFixed(1)} %") ], ) : Column( mainAxisAlignment: MainAxisAlignment.center, children: [ QrImageView(data: myId, size: 250), SizedBox(height: 20), Text("خلّي المرسل يصور الكود"), SizedBox(height: 20), CircularProgressIndicator(), ], ), ), ); } }
